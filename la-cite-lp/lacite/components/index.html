Accessibility

Sass Mixin Source 
//////////////////////////////
// Accessibility Component Mixins
//////////////////////////////
@mixin a11y-hidden($a11y-focusable: $a11y-focusable-default, $extend: $a11y-hidden-base-extend) {
  @if $extend {
    @if $a11y-focusable {
      @extend %a11y-hidden-focusable-extend;
    }
    @else {
      @extend %a11y-hidden-base-extend;
    }
  }
  @else {
    border: 0;
    clip: rect(0 0 0 0);
    height: 1px;
    margin: -1px;
    overflow: hidden;
    padding: 0;
    position: absolute;
    width: 1px;
    @if $a11y-focusable {
      &:active, &:focus {
        clip: auto;
        height: auto;
        margin: 0;
        overflow: visible;
        position: static;
        width: auto;
      }
    }
  }
}

@mixin message-base($extend: $message-base-extend) {
  @if $extend {
    @extend %message-base-extend;
  }
  @else {
    padding: 0.2em 0.5em;
    border: {
      style: solid;
      width: .15em;
      radius: .25em;
    }
  }
}

@mixin message-colors($message-color: $message-bkg-color-default, $message-adjust: $message-color-adjust,  $extend: $message-colors-extend) {
  @if ($message-color != $message-bkg-color-default) {
    $extend: false;
  }
  @if $extend {
    @extend %message-colors-extend;
  }
  @else {
    @include readable-text($message-color);

    @if cs-is-dark($message-color) {
      background-color: blend-screen($message-color, adjust-color($message-color, $lightness: $message-adjust * -1));

      border-color: blend-screen($message-color, adjust-color($message-color, $lightness: $message-adjust * -2.5));
    }
    @else {
      background-color: blend-multiply($message-color, adjust-color($message-color, $lightness: $message-adjust * -1));

      border-color: blend-multiply($message-color, adjust-color($message-color, $lightness: $message-adjust * -2.5));
    }
  }
}

@mixin button-base($extend: $button-base-extend, $make-it-big: true) {
  @if $extend {
    @extend %button-base-extend;
  }
  @else {
    display: inline-block;
    border-radius: 0.15em;
    
    border: 3px solid nth($tertiary-tints,1);
    color: nth($tertiary-tints,1);
    
    font: {
      family: $font-headline;
      weight: normal;
    }
    text: {
      transform: lowercase;
      decoration: none;
    }
    margin: 1em 0;
  }
  
  
}

@mixin button-size-normal($extend: $button-size-normal-extend) {
  @if $extend {
    @extend %button-size-normal-extend;
  }
  @else {
    padding: 0.3em 0.5em;
    font-size: 0.9em;
    line-height: 1.25;
  }
}

@mixin button-size-large($extend: $button-size-large-extend) {
  @if $extend {
    @extend %button-size-large-extend;
  }
  @else {
    padding: 0.5em 0.75em;
    font-size: 1.1em;
    line-height: 1.5;
  }
}

@mixin button-background($button-color: $button-color-default, $button-text-color: $button-text-color-default, $button-adjust: $button-color-adjust) {

  @if $button-text-color == false {
    @include readable-text($button-color);
  }

  @if cs-is-dark($button-color) {
    background-color: blend-screen($button-color, adjust-color($button-color, $lightness: $button-adjust * -1));
  }
  @else {
    background-color: blend-multiply($button-color, adjust-color($button-color, $lightness: $button-adjust * -1));
  }
}

@mixin button-hover($button-color: $button-color-default, $button-text-color: $button-text-color-default, $button-adjust: $button-color-adjust) {
  @if cs-is-dark($button-color) {
    &:hover {
      background-color: blend-screen($button-color, adjust-color($button-color, $lightness: $button-adjust * -.25));
    }
  }
  @else {
    &:hover {
      background-color: blend-multiply($button-color, adjust-color($button-color, $lightness: $button-adjust * -.25));
    }
  }
}

@mixin button-active($button-color: $button-color-default, $button-text-color: $button-text-color-default, $button-adjust: $button-color-adjust) {
  @if cs-is-dark($button-color) {
    &:active {
        background-color: blend-screen($button-color, adjust-color($button-color, $lightness: $button-adjust * 1));
      }
  }
  @else {
    &:active {
        background-color: blend-multiply($button-color, adjust-color($button-color, $lightness: $button-adjust * 1));
      }
  }
}


@mixin button-colors($button-color: $button-color-default, $button-text-color: $button-text-color-default, $button-adjust: $button-color-adjust,  $extend: $button-colors-extend) {
  @if ($button-color != $button-color-default) or ($button-text-color != $button-text-color-default) {
    $extend: false;
  }
  @if $extend {
    @extend %button-colors-extend;
  }
  @else {
    @include button-background($button-color, $button-text-color, $button-adjust);
    @include button-hover($button-color, $button-text-color, $button-adjust);
    @include button-active($button-color, $button-text-color, $button-adjust);
  }
}

@mixin my-button($my-color: nth($primary-tints,1), $im-big: true){
  border: none;
  font-family: inherit;
  font-size: inherit;
  color: inherit;
  background: none;
  cursor: pointer;
  display: inline-block;
  text-transform: uppercase;
  letter-spacing: 1px;
  font-weight: 700;
  outline: none;
  position: relative;
  -webkit-transition: all 0.3s;
  -moz-transition: all 0.3s;
  transition: all 0.3s;
  
  @if $im-big {
   padding: 1.5em 3em; 
  }
  @else {
   padding: 0.75em; 
  }

  &:after {
   content: '';
   position: absolute;
   z-index: -1;
   -webkit-transition: all 0.3s;
   -moz-transition: all 0.3s;
   transition: all 0.3s;
  }
  
  border: 3px solid $my-color;
	 color: $my-color;
  
  &:after {
   width: 0%;
   height: 100%;
   top: 0;
   left: 0;
   background: $my-color;
  }

  &:hover, &:active {
   color: $cs-background;;
  }

  &:hover:after,
  &:active:after {
   width: 100%;
  } 
  
  &:active{
    background-color: shade($my-color, 15);
    border-color: shade($my-color, 15);
    top: 2px;
    left: 1px;
  }
}
